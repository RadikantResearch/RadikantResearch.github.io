Flasher

UART driver for flashing ESP32 family μmicontrollers firmare. Works nicely with NVS.

More Info

Flasher is desktop port of a ESP-Flasher  it supports STUB, and uploads; Application (App), Bootloader (Boot), Partition (Part), Non Volatile Storage (NVS). It really works nicely with the NVS library, because you can generate a in memory NVS partition and upload it to a device. This means in a production line you could easily upload unique data like a serial number, cryptographic certificates to individual units withouth calling a python script.

esp_loader_error_t wrapper_example()
{
    const char *app  = "/Users/charrlie/Binaries/ESP32-C3/app.bin";
    const char *boot = "/Users/charrlie/Binaries/ESP32-C3/boot.bin";
    const char *part = "/Users/charrlie/Binaries/ESP32-C3/part.bin";
    const char *nvs  = "/Users/charrlie/Binaries/ESP32-C3/nvs.bin";

    uint8_t mac_address[6] = {0};
    uint32_t flash_size = 0;

    RETURN_ON_ERROR(wrapper_init(SERIAL_DEVICE_4, 115200));
    RETURN_ON_ERROR(wrapper_flasher_set_file_path(app,boot,part,nvs));   
    RETURN_ON_ERROR(wrapper_get_binaries(ESP32C3_CHIP)); 
    RETURN_ON_ERROR(wrapper_connect_to_stub(ESP32C3_CHIP,115200,HIGHER_BAUD_RATE));
    RETURN_ON_ERROR(wrapper_flasher_upload_binaries(false,false,false,true,MEGA_PAYLOAD));
    RETURN_ON_ERROR(wrapper_get_mac_adress(mac_address));
    RETURN_ON_ERROR(wrapper_get_flash_size(&flash_size));

    // Print in uppercase hexadecimal
    LOGA("The flash_size in hexadecimal : %X", flash_size);
    LOGA("The flash_size in decimal     : %u", flash_size);
    
    // Print the MAC address in hexadecimal format
    LOGA("MAC Address:" CYAN "%02X:%02X:%02X:%02X:%02X:%02X" RESET,
            mac_address[0], mac_address[1], mac_address[2],
            mac_address[3], mac_address[4], mac_address[5]);

    RETURN_ON_ERROR(wrapper_reset_target(1000));
    RETURN_ON_ERROR(wrapper_deinit());

    return ESP_LOADER_SUCCESS;
}
